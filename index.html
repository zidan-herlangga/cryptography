caesar_cipher.py
def caesar_encrypt(text, shift): result = [] for char in text: if char.isalpha(): base = ord('A') if char.isupper() else ord('a') shifted = (ord(char) - base + shift) % 26 + base result.append(chr(shifted)) else: result.append(char) return ''.join(result) def caesar_decrypt(text, shift): return caesar_encrypt(text, -shift) plaintext = "HELLO WORLD" ciphertext = caesar_encrypt(plaintext, 3) print(ciphertext) # KHOOR ZRUOG

def vigenere_encrypt(text, key): result, j = [], 0 key = key.upper() for ch in text: if ch.isalpha(): shift = ord(key[j % len(key)]) - ord('A') base = ord('A') if ch.isupper() else ord('a') result.append(chr((ord(ch)-base+shift) % 26+base)) j += 1 else: result.append(ch) return ''.join(result) def vigenere_decrypt(text, key): result, j = [], 0 key = key.upper() for ch in text: if ch.isalpha(): shift = ord(key[j % len(key)]) - ord('A') base = ord('A') if ch.isupper() else ord('a') result.append(chr((ord(ch)-base-shift) % 26+base)) j += 1 else: result.append(ch) return ''.join(result)

Brute Force pada Caesar Cipher

Mengapa Caesar tidak aman: hanya 26 kemungkinan, attacker bisa mencoba semua dalam hitungan milidetik.

brute_force.py
def brute_force_caesar(ciphertext, hint=None): results = [] for shift in range(26): decrypted = caesar_decrypt(ciphertext, shift) matched = hint and hint.upper() in decrypted.upper() results.append((shift, decrypted, matched)) if matched: print(f"[DITEMUKAN] Shift={shift}: {decrypted}") return results # Kompleksitas: O(26 * n) — sangat cepat! brute_force_caesar("KHOOR ZRUOG", hint="HELLO")
